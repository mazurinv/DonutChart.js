Donut Chart Demo Page

By Vladimir Mazurin GitHub

With custom speed, gradients, header and pointer


          var chart1 = new DonutChart({
              id: 'donut1',
              radius: 100,
              percentage: 60,
              lineWidth: 9,
              pointerWidth: 20,
              lineColor: "#345:rgb(44, 127, 211)",
              backLineColor: "orange:red",
              fontColor: "rgb(44, 127, 211)",
              pointer: './pointer.png',
              header: 'percents',
              speed:.5,
              onRedraw: function() {
                //triggers on animation start
              },
              onAnimationEnd: function() {
                // triggers on animation end
              },
          });

Milestones


          var chart4 = new DonutChart({
              id: 'donut4',
              type: 'milestones',
              radius: 120,
              lineWidth: 5,
              lineColor: "#345:rgb(44, 127, 211)",
              backLineColor: "orange:red",
              fontColor: "rgb(44, 127, 211)",
              pointer: './pointer.png',
              pointerWidth:20,
              speed:.15,
              milestoneImage: './milestone.png',
              milestonesCount: 5,
              nextStep: function(step) {
                // triggers when pointer meets milestone
                console.log(step)
              }
          });
